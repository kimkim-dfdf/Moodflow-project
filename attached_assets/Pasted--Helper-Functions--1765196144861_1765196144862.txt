def get_emotion_by_id(emotion_id):
    """Find an emotion by its ID."""
    for emotion in EMOTIONS:
        if emotion['id'] == emotion_id:
            return emotion
    return None


def get_music_by_emotion(emotion_name, limit):
    """
    Get music recommendations for a specific emotion.
    Returns up to 'limit' songs.
    """
    result = []
    
    for music in MUSIC_RECOMMENDATIONS:
        if music['emotion'] == emotion_name:
            result.append(music)
    
    # Apply limit if specified
    if limit and limit < len(result):
        return result[:limit]
    
    return result


def get_all_book_tags():
    """Get all available book tags."""
    return BOOK_TAGS


def get_books_by_tags(tag_slugs, limit):
    """
    Get books that match ALL of the specified tags (AND logic).
    """
    # If no tags specified, return all books
    if not tag_slugs:
        if limit and limit < len(BOOK_RECOMMENDATIONS):
            return BOOK_RECOMMENDATIONS[:limit]
        return BOOK_RECOMMENDATIONS
    
    # Find books that have ALL specified tags
    result = []
    
    for book in BOOK_RECOMMENDATIONS:
        # Count how many of the requested tags this book has
        match_count = 0
        for tag in tag_slugs:
            if tag in book['tags']:
                match_count = match_count + 1
        
        # Only include if ALL tags match
        if match_count == len(tag_slugs):
            result.append(book)
    
    # Apply limit if specified
    if limit and limit < len(result):
        return result[:limit]
    
    return result


def get_tag_objects_for_book(book):
    """
    Get full tag objects for a book's tags.
    Converts tag slugs to full tag dictionaries.
    """
    result = []
    
    for tag_slug in book['tags']:
        for tag in BOOK_TAGS:
            if tag['slug'] == tag_slug:
                result.append(tag)
    
    return result
